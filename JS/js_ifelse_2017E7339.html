if-else(조건문)

 if조건문은 if() {} else{};가 기본 형태로 조건식에 참,거짓 여부에 따라 실행문을 실행하도록 하는 문장이다.
		1.조건이 한 개인 경우
		 if(조건식) {
			… 실행문 … // 조건식이 참인 경우 
		}
		
		2.조건이 두 개인 경우
		if(조건식) {
			… 실행문1 … // 조건식이 참인 경우 
		}
		else {
			… 실행문2 … // 조건식이 거짓인 경우 
		} 
		
		3.조건이 세 개인 경우
		 if(조건식1) {
		실행문1 // 조건식1이 참인 경우 
		}
		else if(조건식2) {
		실행문2 // 조건식2가 참인 경우 
		}
		…………  

		else {
		실행문n; // 앞의 모든 조건이 거짓인 경우
		}
